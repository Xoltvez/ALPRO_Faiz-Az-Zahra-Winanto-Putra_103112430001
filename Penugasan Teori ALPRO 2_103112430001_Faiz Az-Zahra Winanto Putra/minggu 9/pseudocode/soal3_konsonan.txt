program KategoriHuruf
variabel
    huruf: string
algoritma
    output("Masukan huruf: ")
    input(huruf)
    if huruf == "a" or huruf == "i" or huruf == "u" or huruf == "e" or huruf == "o" or 
       huruf == "A" or huruf == "I" or huruf == "U" or huruf == "E" or huruf == "O" then
        output("Huruf Vokal")
    else
        if (huruf >= "a" and huruf <= "z") or (huruf >= "A" and huruf <= "Z") then
            output("Huruf Konsonan")
        else
            output("Bukan Huruf")
        end if
    end if
end program
