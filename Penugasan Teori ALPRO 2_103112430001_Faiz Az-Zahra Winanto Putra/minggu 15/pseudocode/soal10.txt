program Median
variables
    j, y, number, total: integer

algorithm
    input(y)
    total = 0

    for j = 1 to 9 do
        input(number)
        total = total + number
    end for

    if total >= y * 5 then
        output("Median Value ", y)
    else
        output("Median Value 0")
    end if
end program
