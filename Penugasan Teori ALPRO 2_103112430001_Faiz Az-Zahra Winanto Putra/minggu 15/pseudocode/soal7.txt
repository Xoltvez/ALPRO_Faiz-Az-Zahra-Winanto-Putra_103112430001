program modus
variabel
    x, j, bilangan, nX, nZero: integer

algoritma
    input(x)
    nX = 0
    nZero = 0

    for j = 1 to 9 do
        input(bilangan)
        if bilangan = x maka
            nX = nX + 1
        else if
            nZero = nZero + 1
        end if
    end for

    if nX > nZero maka
        output("Modus = ", x)
    else if
        output("Modus = ", nZero)
    end if
end program
